# Spec: Scope System (Advanced Authentication)

**Status:** Proposed
**Date:** 2026-04-07
**Depends on:** Peregrine core (spec 01), Spectro ORM, Basic Auth (spec 22A), Sessions (spec 19)

---

## 1. Goal

Basic authentication (spec 22A) handles logging users in and out, but it doesn't solve data isolation and authorization at scale. Phoenix solved this with a **scope system** that:

1. **Prevents broken access control** (OWASP Top 10) by scoping database queries to the current user/organization
2. **Integrates with generators** so all generated resources are automatically secure-by-default
3. **Supports multi-tenancy** - Organization-level, team-level, or custom scoping
4. **Enables authorization** - Role-based and permission-based checks with scope awareness

This spec implements **Phoenix's scope system adapted for Swift**, building on the basic authentication from spec 22A.

## 2. Scope

### 2.1 AuthScope Protocol

#### 2.1.1 Protocol Definition

```swift
// In Sources/Peregrine/Auth/Scope.swift

/// Protocol for authentication scopes
/// Scopes carry context about the current request/user and are used for:
/// - Database query scoping (e.g., "only show posts by current user")
/// - Authorization checks (e.g., "user must be admin")
/// - Generator integration (e.g., "auto-add user_id to queries")
public protocol AuthScope: Sendable, Codable {
    /// Unique identifier for this scope type
    static var scopeName: String { get }

    /// Optional: Human-readable description
    static var scopeDescription: String? { get }

    /// The primary identifying value for this scope
    var scopeId: String? { get }

    /// Check if scope is empty (no authenticated entity)
    var isEmpty: Bool { get }
}

extension AuthScope {
    public static var scopeDescription: String? { nil }
    public static var scopeName: String { String(describing: Self.self) }
    public var isEmpty: Bool { scopeId == nil }
}
```

#### 2.1.2 UserScope Implementation

```swift
// In Sources/Peregrine/Auth/UserScope.swift

/// Scope for authenticated users with optional organization context
/// This is the most common scope type for web applications
public struct UserScope: AuthScope {
    public let user: User?
    public let organization: Organization?

    public static var scopeName: String { "user" }
    public static var scopeDescription: String? { "Scope for authenticated users" }

    public var scopeId: String? { user?.id.uuidString }
    public var isEmpty: Bool { user == nil }

    public init(user: User? = nil, organization: Organization? = nil) {
        self.user = user
        self.organization = organization
    }

    /// Create scope for a user
    public static func forUser(_ user: User) -> UserScope {
        UserScope(user: user, organization: nil)
    }

    /// Create scope for a user in an organization
    public static func forUser(_ user: User, organization: Organization) -> UserScope {
        UserScope(user: user, organization: organization)
    }

    /// Add organization to existing scope
    public func withOrganization(_ org: Organization) -> UserScope {
        UserScope(user: self.user, organization: org)
    }

    /// Get scope for testing
    public static func fixture(user: User? = nil, organization: Organization? = nil) -> UserScope {
        UserScope(user: user, organization: organization)
    }
}
```

### 2.2 Scope Configuration

#### 2.2.1 ScopeConfig Metadata

```swift
// In config/scopes.swift (auto-generated by peregrine CLI)

public enum ScopeConfig {
    /// The default scope to use for generators
    public static var defaultScope: String { "user" }

    /// All registered scope types
    public static var scopes: [String: any AuthScope.Type] = [
        "user": UserScope.self,
        "session": SessionScope.self,
    ]

    /// Scope metadata for code generators
    /// Generators use this metadata to create scoped CRUD operations
    public struct ScopeMetadata: Sendable {
        public let name: String                  // e.g., "user"
        public let isDefault: Bool               // Is this the default scope?
        public let assignKey: String             // Assigns key (e.g., "current_scope")
        public let accessPath: [String]          // Path to identifying value (e.g., ["user", "id"])
        public let schemaKey: String              // Foreign key column (e.g., "user_id")
        public let schemaType: String             // Foreign key type (e.g., "UUID" or "Int64")
        public let schemaTable: String?           // Referenced table (e.g., "users")
        public let routePrefix: String?           // Optional route prefix (e.g., "/orgs/:org")
        public let routeAccessPath: [String]?     // Path for route params (e.g., ["organization"])

        public init(
            name: String,
            isDefault: Bool = false,
            assignKey: String = "current_scope",
            accessPath: [String],
            schemaKey: String,
            schemaType: String = "UUID",
            schemaTable: String? = nil,
            routePrefix: String? = nil,
            routeAccessPath: [String]? = nil
        ) {
            self.name = name
            self.isDefault = isDefault
            self.assignKey = assignKey
            self.accessPath = accessPath
            self.schemaKey = schemaKey
            self.schemaType = schemaType
            self.schemaTable = schemaTable
            self.routePrefix = routePrefix
            self.routeAccessPath = routeAccessPath
        }
    }

    /// Metadata for all configured scopes
    public static var scopeMetadata: [String: ScopeMetadata] = [
        "user": ScopeMetadata(
            name: "user",
            isDefault: true,
            assignKey: "current_scope",
            accessPath: ["user", "id"],
            schemaKey: "user_id",
            schemaType: "UUID",
            schemaTable: "users"
        ),
        "session": ScopeMetadata(
            name: "session",
            isDefault: false,
            assignKey: "current_scope",
            accessPath: ["sessionId"],
            schemaKey: "session_id",
            schemaType: "String",
            schemaTable: nil
        ),
    ]

    /// Get metadata for a specific scope
    public static func metadata(for scopeName: String) -> ScopeMetadata? {
        scopeMetadata[scopeName]
    }

    /// Get metadata for default scope
    public static func defaultMetadata() -> ScopeMetadata? {
        metadata(for: defaultScope)
    }
}
```

### 2.3 Scope Loading Plugs

#### 2.3.1 fetchCurrentScope Plug

```swift
// In Sources/Peregrine/Auth/ScopePlugs.swift

/// Load current scope from session or create new scope
/// - Parameter scopeType: Type of scope to load (default: ScopeConfig.defaultScope)
/// - Returns: Plug that loads scope into conn.assigns["current_scope"]
public func fetchCurrentScope(
    for scopeType: String = ScopeConfig.defaultScope
) -> Plug {
    return { conn in
        Task {
            let scope: any AuthScope

            switch scopeType {
            case "user":
                // Try to load user from session
                if let user = try? await conn.loadUser(User.self) {
                    scope = UserScope.forUser(user)
                } else {
                    // Not authenticated - empty scope
                    scope = UserScope()
                }

            case "session":
                // Get or create session ID
                let sessionId = conn.sessionData["id"] as? String ?? UUID().uuidString

                if conn.sessionData["id"] == nil {
                    // Store session ID for future requests
                    var updated = conn
                    updated.putSession("id", sessionId)
                }

                scope = SessionScope(sessionId: sessionId)

            default:
                // Unknown scope type - default to empty user scope
                scope = UserScope()
            }

            // Store scope in assigns for use in routes/contexts
            var updated = conn
            updated.assigns["current_scope"] = scope

            return updated
        }.value
    }
}
```

#### 2.3.2 assignOrgToScope Plug (Optional)

```swift
/// Load organization from route params and add to scope
/// Use this for multi-tenant applications where resources belong to organizations
/// - Returns: Plug that enriches scope with organization context
///
/// Example route: GET /orgs/:org/posts
/// This plug loads the organization from :org param and adds it to current_scope
public func assignOrgToScope() -> Plug {
    return { conn in
        Task {
            // Only process if we have a user scope and org param
            guard let currentScope = conn.assigns["current_scope"] as? UserScope,
                  let orgSlug = conn.params["org"],
                  let user = currentScope.user else {
                return conn
            }

            // Load organization (must belong to current user)
            let repo = conn.assigns[SpectroKey.self] as? SpectroClient

            let org = try? await repo?.query(Organization.self)
                .where(\.slug == orgSlug)
                // Optional: Add organization membership check
                // .where(\.userId == user.id)
                .first()

            var updated = conn

            if let org = org {
                // Enrich scope with organization
                updated.assigns["current_scope"] = currentScope.withOrganization(org)
            } else {
                // Organization not found - could return 404 here
                // For now, continue without organization in scope
            }

            return updated
        }.value
    }
}
```

### 2.4 Authorization Plugs

#### 2.4.1 requireRole Plug

```swift
/// Require specific user role - redirect or forbid if user doesn't have role
/// - Parameters:
///   - role: Required role (e.g., "admin", "moderator")
///   - userType: Model type conforming to Schema (default: User)
///   - redirectIfForbidden: Optional redirect path (otherwise returns 403)
/// - Returns: Plug that enforces role-based authorization
public func requireRole(
    _ role: String,
    userType: any Schema.Type = User.self,
    redirectIfForbidden: String? = nil
) -> Plug {
    return { conn in
        Task {
            guard let scope = conn.assigns["current_scope"] as? UserScope,
                  let user = scope.user else {
                return redirectIfForbidden.map { Response.redirect(to: $0) } ?? conn
            }

            // Check if user has role (requires User.role field)
            let userRole = (user as? UserRoleProvider)?.role ?? "user"

            if userRole != role && userRole != "admin" {
                // Admins have all roles
                return redirectIfForbidden.map { Response.redirect(to: $0) }
                    ?? Response(status: .forbidden)
            }

            return conn
        }.value
    }
}

/// Protocol for entities that can provide role information
public protocol UserRoleProvider {
    var role: String { get }
}
```

#### 2.4.2 requireOwnership Plug

```swift
/// Require ownership of a resource - fail if resource doesn't belong to current user
/// - Parameters:
///   - resourceType: Type of resource to check
///   - idParam: Route param containing resource ID (default: "id")
///   - userIdKey: Property on resource that contains user ID (default: "userId")
///   - userType: Type of user model (default: User)
/// - Returns: Plug that enforces resource ownership
///
/// Example: GET /posts/:id where posts.userId must equal current user.id
public func requireOwnership<T: Schema>(
    _ resourceType: T.Type,
    idParam: String = "id",
    userIdKey: String = "userId",
    userType: any Schema.Type = User.self
) -> Plug {
    return { conn in
        Task {
            guard let currentUser = conn.currentUser(userType),
                  let resourceId = conn.params[idParam] else {
                return Response(status: .forbidden)
            }

            let repo = conn.assigns[SpectroKey.self] as? SpectroClient

            // Load resource
            guard let resource = try? await repo?.query(resourceType)
                .where(\.id == UUID(uuidString: resourceId!)!)
                .first() else {
                return Response(status: .notFound)
            }

            // Check ownership
            let resourceUserId = resource.value(forKey: userIdKey) as? UUID
            let currentUserId = currentUser.value(forKey: "id") as? UUID

            if resourceUserId != currentUserId {
                return Response(status: .forbidden)
            }

            // Inject resource into assigns for use in route handler
            var updated = conn
            updated.assigns["current_resource"] = resource

            return updated
        }.value
    }
}
```

#### 2.4.3 requirePermission Plug

```swift
/// Require specific permission - fail if user lacks permission
/// - Parameters:
///   - permission: Permission string (e.g., "posts.create", "users.delete")
///   - userType: Type of user model (default: User)
///   - redirectIfForbidden: Optional redirect path (otherwise returns 403)
/// - Returns: Plug that enforces permission-based authorization
public func requirePermission(
    _ permission: String,
    userType: any Schema.Type = User.self,
    redirectIfForbidden: String? = nil
) -> Plug {
    return { conn in
        Task {
            guard let user = conn.currentUser(userType) else {
                return redirectIfForbidden.map { Response.redirect(to: $0) }
                    ?? Response(status: .forbidden)
            }

            // Check if user has permission
            guard let permissionProvider = user as? PermissionProvider else {
                return redirectIfForbidden.map { Response.redirect(to: $0) }
                    ?? Response(status: .forbidden)
            }

            if !permissionProvider.hasPermission(permission) {
                return redirectIfForbidden.map { Response.redirect(to: $0) }
                    ?? Response(status: .forbidden)
            }

            return conn
        }.value
    }
}

/// Protocol for entities that can check permissions
public protocol PermissionProvider {
    func hasPermission(_ permission: String) -> Bool
}
```

### 2.5 Scope-Aware Contexts

#### 2.5.1 Context Pattern

Contexts should accept a `scope` parameter and use it to filter queries:

```swift
// In Sources/MyApp/Contexts/PostsContext.swift
import Peregrine
import SpectroKit

struct PostsContext {
    let conn: Connection
    let scope: any AuthScope

    /// List all posts scoped to current user
    func listPosts() async throws -> [Post] {
        let repo = conn.assigns[SpectroKey.self] as! SpectroClient
        let userScope = scope as! UserScope

        // Automatically scope query to current user
        return try await repo.query(Post.self)
            .where(\.userId == userScope.user!.id)
            .all()
    }

    /// Get a specific post by ID, scoped to current user
    func getPost(id: UUID) async throws -> Post {
        let repo = conn.assigns[SpectroKey.self] as! SpectroClient
        let userScope = scope as! UserScope

        guard let post = try await repo.query(Post.self)
            .where(\.userId == userScope.user!.id)
            .where(\.id == id)
            .first() else {
            throw AbortError(.notFound)
        }

        return post
    }

    /// Create a new post scoped to current user
    func createPost(_ post: Post) async throws -> Post {
        let repo = conn.assigns[SpectroKey.self] as! SpectroClient
        let userScope = scope as! UserScope

        var created = post
        created.userId = userScope.user!.id

        return try await repo.save(created)
    }

    /// Update a post (automatically scoped)
    func updatePost(_ post: Post) async throws -> Post {
        let repo = conn.assigns[SpectroKey.self] as! SpectroClient

        var updated = post
        updated.updatedAt = Date()

        return try await repo.save(updated)
    }

    /// Delete a post (automatically scoped)
    func deletePost(_ post: Post) async throws {
        let repo = conn.assigns[SpectroKey.self] as! SpectroClient

        try await repo.delete(post)
    }

    /// List posts for specific organization (if in org scope)
    func listPostsForOrganization() async throws -> [Post] {
        let repo = conn.assigns[SpectroKey.self] as! SpectroClient
        let userScope = scope as! UserScope

        var query = repo.query(Post.self)
            .where(\.userId == userScope.user!.id)

        // If we have an organization in scope, filter by it
        if let org = userScope.organization {
            query = query.where(\.organizationId == org.id)
        }

        return try await query.all()
    }
}
```

#### 2.5.2 Route Integration

```swift
// In Sources/MyApp/Routes/PostsRoutes.swift
extension PeregrineApp {
    var postsRoutes: [Route] {
        [
            GET("/posts") { conn in
                // Scope is automatically loaded by fetchCurrentScope plug
                let context = PostsContext(
                    conn: conn,
                    scope: conn.assigns["current_scope"]!  // Force unwrap - always set
                )

                let posts = try await context.listPosts()
                return Response.render("posts/index", ["posts": posts])
            },

            GET("/posts/:id") { conn in
                let context = PostsContext(
                    conn: conn,
                    scope: conn.assigns["current_scope"]!
                )

                let post = try await context.getPost(id: conn.params["id"]!)
                return Response.render("posts/show", ["post": post])
            },

            POST("/posts") { conn in
                let context = PostsContext(
                    conn: conn,
                    scope: conn.assigns["current_scope"]!
                )

                var post = try conn.decode(Post.self)
                let created = try await context.createPost(post)
                return Response.redirect(to: "/posts/\(created.id)")
            },

            PUT("/posts/:id") { conn in
                let context = PostsContext(
                    conn: conn,
                    scope: conn.assigns["current_scope"]!
                )

                var post = try await context.getPost(id: conn.params["id"]!)
                post = try conn.decode(Post.self)
                let updated = try await context.updatePost(post)
                return Response.redirect(to: "/posts/\(updated.id)")
            },

            DELETE("/posts/:id") { conn in
                let context = PostsContext(
                    conn: conn,
                    scope: conn.assigns["current_scope"]!
                )

                let post = try await context.getPost(id: conn.params["id"]!)
                try await context.deletePost(post)
                return Response.redirect(to: "/posts")
            }
        ]
    }
}
```

### 3.1 Basic User Scoping

```swift
// In Sources/MyApp/App.swift
var plugs: [Plug] {
    [
        session(store: .postgres),
        fetchCurrentScope(),  // Load scope into conn.assigns["current_scope"]

// All routes now have access to current_scope

    let user = scope.user!

```

### 3.2 Multi-Tenant Organization Scoping

```swift
var plugs: [Plug] {
    [
        session(store: .postgres),
        fetchCurrentScope(),
        assignOrgToScope(),  // Add org to scope from :org param
        router()
    ]
}

// Routes like GET /orgs/:slug/posts now have org in scope
scope("/orgs/:slug", through: [requireAuth()]) {
    GET("/posts") { conn in
        let scope = conn.assigns["current_scope"] as! UserScope
        let context = PostsContext(conn: conn, scope: scope)

        // Posts are automatically scoped to:
        // 1. Current user (userId == user.id)
        // 2. Current organization (organizationId == org.id)
        let posts = try await context.listPostsForOrganization()

        return Response.render("posts/index", [
            "posts": posts,
            "organization": scope.organization
        ])
    }
}
```

## 4. Acceptance Criteria

### 4.1 AuthScope Protocol
- [ ] `AuthScope` protocol defined with required properties
- [ ] `UserScope` implementation supports user and organization
- [ ] `SessionScope` implementation for anonymous sessions
- [ ] `isEmpty` property checks if scope has authenticated entity
- [ ] All scope types are Sendable and Codable

### 4.2 Scope Configuration
- [ ] `ScopeConfig` holds scope metadata in `config/scopes.swift`
- [ ] `ScopeMetadata` includes: name, assignKey, accessPath, schemaKey, schemaType, schemaTable, routePrefix
- [ ] `ScopeConfig.defaultScope` returns default scope name
- [ ] `ScopeConfig.metadata(for:)` returns metadata for specific scope
- [ ] Multiple scopes can be configured simultaneously
- [ ] One scope can be marked as default

### 4.3 Scope Loading
- [ ] `fetchCurrentScope()` plug loads scope into `conn.assigns["current_scope"]`
- [ ] `fetchCurrentScope()` loads user from session for "user" scope
- [ ] `fetchCurrentScope()` creates session ID for "session" scope
- [ ] `assignOrgToScope()` plug enriches scope with organization from route params
- [ ] Scope loading works with session system (spec 19)
- [ ] Empty scopes are created for unauthenticated requests

### 4.4 Authorization Plugs
- [ ] `requireRole()` plug checks user role, redirects or forbids if insufficient
- [ ] `requireRole()` allows admins to bypass role checks
- [ ] `requireOwnership()` plug validates resource ownership by current user
- [ ] `requireOwnership()` returns 403 if ownership check fails
- [ ] `requirePermission()` plug checks custom permissions
- [ ] `UserRoleProvider` protocol for role-based authorization
- [ ] `PermissionProvider` protocol for permission-based authorization

### 4.5 Context Integration
- [ ] Contexts accept `scope` parameter
- [ ] Context methods use `scope` to filter database queries
- [ ] Context methods automatically add `userId` to created records
- [ ] Context methods automatically filter by `userId` in read operations
- [ ] `getPost()` throws 404 if resource doesn't belong to current user
- [ ] Contexts work with UserScope and UserScope+Organization

### 4.6 Generator Integration
- [ ] Generators use `ScopeConfig` metadata for code generation
- [ ] Generated schemas include `user_id` foreign key
- [ ] Generated contexts use scope parameter
- [ ] Generated routes pass scope to contexts
- [ ] Generated migrations include `user_id` constraint
- [ ] Generated tests verify user scoping works correctly

### 4.7 Testing Support
- [ ] `UserScope.fixture()` creates test scopes
- [ ] `SessionScope.fixture()` creates test session scopes
- [ ] TestApp provides scope-aware test helpers
- [ ] Can test multi-tenant scenarios
- [ ] Can test role-based authorization
- [ ] Can test permission-based authorization

### 4.8 Error Handling
- [ ] Missing scope in assigns is handled gracefully (use default)
- [ ] Failed organization queries don't crash request
- [ ] Ownership checks return proper HTTP status codes
- [ ] Permission checks return proper HTTP status codes

## 5. Non-goals

- No built-in user model (implement at application level)
- No built-in organization model (implement at application level)
- No automatic permission system (implement PermissionProvider at application level)
- No role management UI (implement at application level)
- No policy-based authorization (use PermissionProvider protocol)
- No attribute-based access control (ABAC)
- No hierarchical roles (use simple string comparison)
- No team-based scoping (can add custom scope types)
- No caching of scope lookups (optimize in production if needed)
- No distributed scope synchronization (use database for now)

## 7. Migration Notes

This spec builds on spec 22A (Basic Auth). Migration guide:

### For New Apps
1. Start with spec 22A (basic authentication)
2. Add `fetchCurrentScope()` to your plug pipeline
3. Update contexts to accept `scope` parameter
4. Use generators to create scoped CRUD resources

### For Existing Apps (with Basic Auth)
1. Add `fetchCurrentScope()` plug after `session()` plug
2. Update route handlers to use `conn.assigns["current_scope"]`
3. Update contexts to accept and use scope parameter
4. Gradually add authorization plugs (`requireRole`, `requirePermission`)

**Before (Basic Auth):**
```swift
GET("/posts") { conn in
    let user = conn.currentUser(User.self)!
    let repo = conn.assigns[SpectroKey.self] as! SpectroClient
    let posts = try await repo.query(Post.self)
        .where(\.userId == user.id)  // Manual scoping
        .all()
    return Response.render("posts/index", ["posts": posts])
}
```

**After (with Scopes):**
```swift
GET("/posts") { conn in
    let context = PostsContext(
        conn: conn,
        scope: conn.assigns["current_scope"]!  // Auto-scoped
    )
    let posts = try await context.listPosts()  // Automatic scoping
    return Response.render("posts/index", ["posts": posts])
}
```

## 8. Performance Considerations

- **Scope loading** - Single database query per request (cached in session)
- **Query scoping** - Adds `WHERE userId = ?` to all queries (indexed)
- **Organization loading** - Additional query if using `assignOrgToScope()`
- **Context creation** - Cheap (just stores references, no queries)
- **Authorization checks** - In-memory (no database queries after user loaded)

**Optimization Tips:**
- Add database indexes on `user_id` and `organizationId` columns
- Use connection pooling for high-traffic applications
- Cache frequently-accessed scope metadata in production

---

## 9. Security Considerations

- **Query scoping** - All queries automatically filter by current user's ID
- **Organization isolation** - Multi-tenant apps can't access other organizations' data
- **Authorization layers** - Auth (who are you?) + Scope (what can you access?) + Permissions (what are you allowed to do?)
- **403 vs 404** - Ownership checks return 403 (forbidden), not 404 (not found) to prevent data enumeration
- **Admin bypass** - Admins have all roles, but don't bypass ownership checks
- **Scope hijacking** - Scopes are loaded from session, not from user input

## 10. Future Enhancements

Possible follow-up features:

- **Team scoping** - Add TeamScope for team-based access control
- **Hierarchical scopes** - Support nested organizations/teams
- **Scope caching** - Cache scope lookups in production for performance
- **Policy engine** - Integrate policy-based authorization systems
- **Audit logging** - Log all authorization decisions for compliance
- **Scope middleware** - Chain multiple scope enrichments
- **Dynamic scopes** - Scopes that change based on request context

```swift
import Testing

struct ScopeSystemTests {

    func userScopeLoads() async throws {

    @Test("scope provides query isolation")
    func queryIsolation() async throws {
        let app = TestApp(MyApp())
        let repo = app.database

        // Create two users
        var user1 = User()
        user1.email = "user1@example.com"
        user1.hashedPassword = try Auth.hashPassword("password")

        user2.email = "user2@example.com"

        var post1 = Post()
        post1.userId = saved1.id
        post1.title = "User 1's Post"
        try await repo.save(post1)

        var post2 = Post()
        post2.userId = saved2.id
        post2.title = "User 2's Post"
        try await repo.save(post2)

        // Query posts as user1
        var conn1 = app.authenticatedAs(saved1)
        let context1 = PostsContext(
            conn: conn1,
            scope: UserScope.forUser(saved1)
        )

        let posts1 = try await context1.listPosts()

        // Should only see user1's posts
        #expect(posts1.count == 1)
        #expect(posts1[0].id == post1.id)
    }

    @Test("requireOwnership allows only owner")
    func requireOwnershipWorks() async throws {
        let app = TestApp(MyApp())
        let repo = app.database

        // Create users and post
        var user1 = User()
        user1.email = "user1@example.com"
        user1.hashedPassword = try Auth.hashPassword("password")
        let saved1 = try await repo.save(user1)

        var post = Post()
        post.userId = saved1.id
        post.title = "User 1's Post"
        let saved = try await repo.save(post)

        // Try to access as user1 (owner)
        var conn1 = app.authenticatedAs(saved1)
        conn1.params["id"] = saved.id.uuidString

        let plug = requireOwnership(Post.self)
        let result1 = plug(conn1)

        // Should succeed
        #expect(result1.status == .ok)

        // Try to access as user2 (not owner)
        var user2 = User()
        user2.email = "user2@example.com"
        user2.hashedPassword = try Auth.hashPassword("password")
        let saved2 = try await repo.save(user2)

        var conn2 = app.authenticatedAs(saved2)
        conn2.params["id"] = saved.id.uuidString

        let result2 = plug(conn2)

        // Should fail
        #expect(result2.status == .forbidden)
    }
}
```
